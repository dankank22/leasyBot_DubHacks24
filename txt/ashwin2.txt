import java.io.*;

public class TwitterTrends {
    private TweetBot bot;

    public TwitterTrends(TweetBot bot){
        this.bot = bot;
    }

    // analyzes each tweet in the TweetBot and returns that word that occurs the most in all tweets
    // Returns:
    //    most common occuring word (String)
    public String getMostFrequentWord(){
        Map<String, Integer> freqMap = new HashMap<>();
        for(int i = 0; i < bot.numTweets(); i++){
            String tweet = bot.nextTweet();
            Scanner tweetScan = new Scanner(tweet);
            while(tweetScan.hasNext()){
                String word = tweetScan.next().toLowerCase();
                if(!freqMap.containsKey(word)){
                    freqMap.put(word, 1);
                } else {
                    freqMap.put(word, freqMap.get(word)+1);
                }
            }
        }
        int max = 0;
        String mostFreqWord = "";
        for(String word : freqMap.keySet()){
            if(freqMap.get(word) > max){
                max = freqMap.get(word);
                mostFreqWord = word;
            }
        }
        return mostFreqWord;
    }

    // Creative element: Scans through all tweets and prints a disclaimer when there is 
    // potential misinformation about the COVID-19 pandemic.
    public void printCdcDisclaimer(){
        System.out.println("Scanning tweets for misinformation...");

    }

    public static Set<String> pruneDictionary(List<String> contents, int wordLength) {

        Set<String> words = new TreeSet<String>();

    // Handles the users guess, finds out the set with most words and returns the pattern that

    public static String record(String guess, Set<String> words, int wordLength) {
        if(words.size() == 0 || guess.length() != wordLength){
            throw new IllegalArgumentException("Dictionary size is zero, or guess has wrong size");
        }

        Map<String, Set<String>> possiblePatterns = new TreeMap<>();

            if(possiblePatterns.containsKey(pattern)){
                possiblePatterns.get(pattern).add(word);

                Set<String> patternSet = new TreeSet<>();

                possiblePatterns.put(pattern, patternSet);
            }

        while(itr.hasNext()){

                itr.remove();
            }
        }

    }

    // Develops the pattern given a possible word in the set and the users guess
    // Parameters:
    //     User's guess (String)
    //     word from the set of words (String)
    // Returns:
    //     The pure(wordle) pattern for matching the 2 words (String)
    public static String patternFor(String word, String guess) {
        String result = "";
        for(int i = 0; i < guess.length(); i++){
            if(guess.charAt(i) == word.charAt(i)){
                guess = guess.substring(0,i) + "!" + guess.substring(i+1);
                word = word.substring(0,i) + "!" + word.substring(i+1);
            }
        }

        for(int i = 0; i < guess.length(); i++){

            String letterString = ""+letter;
            if(Character.isLetter(letter) && guess.contains(letterString)){
                guess = guess.replaceFirst(letterString, "@");
                word = word.replaceFirst(letterString, "@");

                word.replace(letterString, "%");
            }
        }

        for(int i = 0; i < guess.length(); i++){
            if(guess.charAt(i) == '!'){
                result += GREEN;
            } else if(guess.charAt(i) == '@'){
                result += YELLOW;
            } else {
                result += GRAY;
            }
        }
        
        return result;
    }
    
    // Helper method
    // Given all possibilities, returns the pattern that has the most word options.
    // Parameters:

    public static String findMaxPattern(Map<String, Set<String>> patternSet){

import java.util.*;
import java.io.*;

// This class represents a method to compress and decompress files deviced by David Huffman
// the compressions will be lossless.
public class HuffmanCode {

    private HuffmanNode root; 

    // Creates a new HuffmanCode object with an input array 
    // Input array contains count of character as the value and index 
    // is the ascii value of that character
    // Character does not get included if it has zero or negative frequency
    // Parameters:
    //    frequencies - input array (int[])
    public HuffmanCode(int[] frequencies){
        PriorityQueue<HuffmanNode> pq = new PriorityQueue<>();
        for(int i = 0; i < frequencies.length; i++){
            if(frequencies[i] > 0){
                HuffmanNode node = new HuffmanNode(i, frequencies[i]);
                pq.add(node);
            }
        }

        while (pq.size() > 1) {
            HuffmanNode left = pq.remove();
            HuffmanNode right = pq.remove();
            HuffmanNode parent = new HuffmanNode(-1, left.frequency+ right.frequency, left, right);
            pq.add(parent);
        }

        root = pq.remove();
    }

    // Constructs a HuffmanCode object with help of a previously saved input .code file
    // in out file has first line as the ascii value of character and next line as the 
    // huffman coding
    // Parameter:
    //    Scanner object for the input .code file
    public HuffmanCode(Scanner input){
        this.root = new HuffmanNode(-1, -1);
        while(input.hasNextLine()){
            String asciiStr = input.nextLine();
            int ascii = Integer.parseInt(asciiStr);
            String binaryCode = input.nextLine();

            root = HuffmanCode(root, ascii, binaryCode);
        }
    }

    private HuffmanNode HuffmanCode(HuffmanNode root, int ascii, String code){
        HuffmanNode node = root;
        if(code.length() == 0){
            node.ascii = ascii;
        } else {
            if(code.substring(0, 1).equals("1")){
                // System.out.println("entered right");
                if(root.right == null){
                    root.right = new HuffmanNode(-1, -1);
                }
                node.right = HuffmanCode(root.right, ascii, code.substring(1));
            } else {
                // System.out.println("entered left");
                if(root.left == null){
                    root.left = new HuffmanNode(-1, -1);
                }
                node.left = HuffmanCode(root.left, ascii, code.substring(1));
            }
        }

        return node;
    }

    // Saves the current HuffmanCode to a output file 
    // in the same format as constructor 2 (standard format)
    // Parameters:
    //    PrintStream object to the output file
    public void save(PrintStream output){
        save(output, root, "");
    }

    private void save(PrintStream output, HuffmanNode root, String code){
        if(root.ascii != -1){
            output.println(root.ascii);
            output.println(code);
        } else{
            save(output, root.left, code + "0");
            save(output, root.right, code + "1");
        }
    }

    // Reads individual bits from a input stream, and writes the characters
    // to the output file. Stops reading when the input file is empty
    // Parameters:
    //    input file of encoded characters (BitInputStream)
    //    output file to write to (PrintStream)
    public void translate(BitInputStream input, PrintStream output){
        HuffmanNode node = root;
        while (input.hasNextBit()) {
            int bit = input.nextBit();
            if(bit == -1){
                return;
            }
            if (bit == 0) {
                node = node.left;
            } else {
                node = node.right;
            }

            if (node.left == null && node.right == null) {
                output.write((char) node.ascii);
                node = root;
            }
        }
    }

    // This class represents a node in the HuffmanCode tree
    // implements the Comparable interface to order HuffmanNodes
    private static class HuffmanNode implements Comparable<HuffmanNode>{
        private int ascii;
        private int frequency;
        private HuffmanNode left;
        private HuffmanNode right;

        // Constructs a HuffmanNode object with (parameters:)an ascii code and frequency
        public HuffmanNode(int code, int freq) {
            this(code, freq, null, null);
        }

        // Constructs a HuffmanNode object with (parameters:)an ascii code and frequency
        // and the next children of this node
        public HuffmanNode(int code, int freq, HuffmanNode left, HuffmanNode right) {
            this.ascii = code;
            this.frequency = freq;
            this.left = left;
            this.right = right;
        }

        // Compares this HuffmanNode object to another open
        // parameters: the other HuffmanNode object to compare with
        // returns : difference between the two nodes
        @Override
        public int compareTo(HuffmanNode other) {
            return this.frequency - other.frequency;
        }
    }
}

Fast Retrieval: O(1) average case time complexity for lookups, making it efficient for accessing data.
Dynamic Size: Can dynamically resize to accommodate the growing number of songs.
Simple Implementation: Easy to implement and use.

Binary Search Tree (BST):
Pros:

Efficient Search: O(log n) average case time complexity for lookups.
Cons:

Imbalanced Tree: Without balancing, the tree can become skewed, leading to worst-case scenarios of O(n) for search operations.

Best Case: O(log n) (balanced tree)
Worst Case: O(n) (unbalanced tree)

AVL Tree:
Pros:

Balanced: Guarantees a balanced tree, ensuring O(log n) time complexity for search operations.
Ordered Storage: Maintains a sorted order.
Cons:

Complexity: More complex to implement than a basic BST.
Overhead: Requires additional bookkeeping to maintain balance.
Best Case: O(log n)
Worst Case: O(log n)

ArrayList:
Pros:

Fast Access: O(1) time complexity for direct access using indices.
Dynamic Sizing: Can dynamically resize to accommodate the growing number of songs.
Cons:

Insertion/Deletion Overhead: O(n) time complexity for inserting or deleting elements in the middle, as it may require shifting elements.
Best Case: O(1) (for direct access)
Worst Case: O(n) (for insertion or deletion in the middle)

LinkedList:

Efficient Insertion/Deletion: O(1) time complexity for inserting or deleting elements, assuming you have a reference to the node.
Dynamic Sizing: Can dynamically resize.
Cons:

Sequential Access: O(n) time complexity for random access, as you may need to traverse the list.
Best Case: O(1) (for insertion/deletion with a reference)
Worst Case: O(n) (for random access)

Choose the data structure based on the specific requirements and operations in your use case.
